19 auto MarketOrderBook::onMarketUpdate(const Exchange::MEMarketUpdate *market_update) noexcept -> void {
20 const auto bid_updated = (bids_by_price_ && market_update->side_ == Side::BUY && market_update->price_ >= bids_by_price_->price_);
21 const auto ask_updated = (asks_by_price_ && market_update->side_ == Side::SELL && market_update->price_ <= asks_by_price_->price_);
22
23 switch (market_update->type_) {
24 case Exchange::MarketUpdateType::ADD: {
25 auto order = order_pool_.allocate(market_update->order_id_, market_update->side_, market_update->price_,
26 market_update->qty_, market_update->priority_, nullptr, nullptr);
27 addOrder(order);
28 }
29 break;
30 case Exchange::MarketUpdateType::MODIFY: {
31 auto order = oid_to_order_.at(market_update->order_id_);
32 order->qty_ = market_update->qty_;
33 }
34 break;
35 case Exchange::MarketUpdateType::CANCEL: {
36 auto order = oid_to_order_.at(market_update->order_id_);
37 removeOrder(order);
38 }
39 break;
40 case Exchange::MarketUpdateType::TRADE: {
41 trade_engine_->onTradeUpdate(market_update, this);
42 return;
43 }
44 break;
45 case Exchange::MarketUpdateType::CLEAR: {
46 for (auto &order: oid_to_order_) {
47 if (order)
48 order_pool_.deallocate(order);
49 }
50 oid_to_order_.fill(nullptr);
51
52 if(bids_by_price_) {
53 for(auto bid = bids_by_price_->next_entry_; bid != bids_by_price_; bid = bid->next_entry_)
54 orders_at_price_pool_.deallocate(bid);
55 orders_at_price_pool_.deallocate(bids_by_price_);
56 }
57
58 if(asks_by_price_) {

62 }

64 bids_by_price_ = asks_by_price_ = nullptr;
65 }
66 break;
67 case Exchange::MarketUpdateType::INVALID:
68 case Exchange::MarketUpdateType::SNAPSHOT_START:
69 case Exchange::MarketUpdateType::SNAPSHOT_END:
70 break;
71 }
72
73 updateBBO(bid_updated, ask_updated);
74
75 logger_->log("%:% %() % % %", __FILE__, __LINE__, __FUNCTION__,
76 Common::getCurrentTimeStr(&time_str_), market_update->toString(), bbo_.toString());
