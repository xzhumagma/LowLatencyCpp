166 auto MarketDataConsumer::recvCallback(McastSocket *socket) noexcept -> void {
167 const auto is_snapshot = (socket->socket_fd_ == snapshot_mcast_socket_.socket_fd_);
168 if (UNLIKELY(is_snapshot && !in_recovery_)) { // market update was read from the snapshot market data stream and we are not in recovery, so we dont need it and discard it.
169 socket->next_rcv_valid_index_ = 0;
170
171 logger_.log("%:% %() % WARN Not expecting snapshot messages.\n",
172 __FILE__, __LINE__, __FUNCTION__, Common::getCurrentTimeStr(&time_str_));
173
174 return;
175 }
176 // go through the socket->rcv_buffer_buffer and read it in chunks of sizeof(Exchange::MDPMarketUpdate) bytes.
177 // the goal here is to read as many full MDP market data update messages as possible from the socket->rcv_buffer_buffer.
178 // convert thedata in the buffer to an object of the Exchange::MDPMarketUpdate type.
179 if (socket->next_rcv_valid_index_ >= sizeof(Exchange::MDPMarketUpdate)) {
180 size_t i = 0;
181 for (; i + sizeof(Exchange::MDPMarketUpdate) <= socket->next_rcv_valid_index_; i += sizeof(Exchange::MDPMarketUpdate)) {
182 // vector.data()is a member function that returns the direct pointer to the memory arrray used internally by the vector
183 // to store its owned elements.
184 // the read it in chunks of sizeof(Exchange::MDPMarketUpdate) bytes.
185 auto request = reinterpret_cast<const Exchange::MDPMarketUpdate *>(socket->inbound_data_.data() + i);
186 logger_.log("%:% %() % Received % socket len:% %\n", __FILE__, __LINE__, __FUNCTION__,
187 Common::getCurrentTimeStr(&time_str_),
188 (is_snapshot ? "snapshot" : "incremental"), sizeof(Exchange::MDPMarketUpdate), request->toString());
189 // whether if there is a sequence number gap or whether we are already in recovery.
190 const bool already_in_recovery = in_recovery_;
191 in_recovery_ = (already_in_recovery || request->seq_num_ != next_exp_inc_seq_num_);
192 // in_recovery is a rare case: the case that we need to handle it.
193 if (UNLIKELY(in_recovery_)) {
194 if (UNLIKELY(!already_in_recovery)) { // if we just entered recovery, start the snapshot synchonization process by subscribing to the snapshot multicast stream.
195 logger_.log("%:% %() % Packet drops on % socket. SeqNum expected:% received:%\n", __FILE__, __LINE__, __FUNCTION__,
196 Common::getCurrentTimeStr(&time_str_), (is_snapshot ? "snapshot" : "incremental"), next_exp_inc_seq_num_, request->seq_num_);
197 startSnapshotSync();
198 }
199
200 queueMessage(is_snapshot, request); // queue up the market data update message and check if snapshot recovery / synchronization can be completed successfully.
201 } else if (!is_snapshot) { // not in recovery and received a packet in the correct order and without gaps, process it.
202 logger_.log("%:% %() % %\n", __FILE__, __LINE__, __FUNCTION__,
203 Common::getCurrentTimeStr(&time_str_), request->toString());
204
205 ++next_exp_inc_seq_num_;
206
207 auto next_write = incoming_md_updates_->getNextToWriteTo();
208 *next_write = std::move(request->me_market_update_);
209 incoming_md_updates_->updateWriteIndex();
210 }
211 }
212 memcpy(socket->inbound_data_.data(), socket->inbound_data_.data() + i, socket->next_rcv_valid_index_ - i);
213 socket->next_rcv_valid_index_ -= i;
214 }
