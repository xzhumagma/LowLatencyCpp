27 auto SnapshotSynthesizer::addToSnapshot(const MDPMarketUpdate *market_update) {
28 const auto &me_market_update = market_update->me_market_update_;
29 auto *orders = &ticker_orders_.at(me_market_update.ticker_id_);
30 switch (me_market_update.type_) {
31 case MarketUpdateType::ADD: {
32 auto order = orders->at(me_market_update.order_id_);
33 ASSERT(order == nullptr, "Received:" + me_market_update.toString() + " but order already exists:" + (order ? order->toString() : ""));
34 orders->at(me_market_update.order_id_) = order_pool_.allocate(me_market_update);
35 }
36 break;
37 case MarketUpdateType::MODIFY: {
38 auto order = orders->at(me_market_update.order_id_);
39 ASSERT(order != nullptr, "Received:" + me_market_update.toString() + " but order does not exist.");
40 ASSERT(order->order_id_ == me_market_update.order_id_, "Expecting existing order to match new one.");
41 ASSERT(order->side_ == me_market_update.side_, "Expecting existing order to match new one.");
42
43 order->qty_ = me_market_update.qty_;
44 order->price_ = me_market_update.price_;
45 }
46 break;
47 case MarketUpdateType::CANCEL: {
48 auto order = orders->at(me_market_update.order_id_);
49 ASSERT(order != nullptr, "Received:" + me_market_update.toString() + " but order does not exist.");
50 ASSERT(order->order_id_ == me_market_update.order_id_, "Expecting existing order to match new one.");
51 ASSERT(order->side_ == me_market_update.side_, "Expecting existing order to match new one.");
52
53 order_pool_.deallocate(order);
54 orders->at(me_market_update.order_id_) = nullptr;
55 }
56 break;
57 case MarketUpdateType::SNAPSHOT_START:
58 case MarketUpdateType::CLEAR:
59 case MarketUpdateType::SNAPSHOT_END:
60 case MarketUpdateType::TRADE:
61 case MarketUpdateType::INVALID:
62 break;
63 }
64
65 ASSERT(market_update->seq_num_ == last_inc_seq_num_ + 1, "Expected incremental seq_nums to increase.");
66 last_inc_seq_num_ = market_update->seq_num_;
67 }
