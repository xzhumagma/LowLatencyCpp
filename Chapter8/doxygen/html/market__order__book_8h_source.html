147 auto removeOrdersAtPrice(Side side, Price price) noexcept {
148 const auto best_orders_by_price = (side == Side::BUY ? bids_by_price_ : asks_by_price_);
149 auto orders_at_price = getOrdersAtPrice(price);

169 auto removeOrder(MarketOrder *order) noexcept -> void {

172 if (order->prev_order_ == order) { // only one element.
