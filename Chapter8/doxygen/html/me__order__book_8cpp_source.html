54 auto MEOrderBook::checkForMatch(ClientId client_id, OrderId client_order_id, TickerId ticker_id, Side side, Price price, Qty qty, Qty new_market_order_id) noexcept {
55 auto leaves_qty = qty;
56
57 if (side == Side::BUY) {
58 while (leaves_qty && asks_by_price_) {
59 const auto ask_itr = asks_by_price_->first_me_order_;
60 if (LIKELY(price < ask_itr->price_)) {

62 }

64 match(ticker_id, client_id, side, client_order_id, new_market_order_id, ask_itr, &leaves_qty);
65 }

67 if (side == Side::SELL) {
68 while (leaves_qty && bids_by_price_) {

74 match(ticker_id, client_id, side, client_order_id, new_market_order_id, bid_itr, &leaves_qty);
75 }
76 }

148 orderIdToString(o_itr->market_order_id_).c_str(), qtyToString(o_itr->qty_).c_str(),
