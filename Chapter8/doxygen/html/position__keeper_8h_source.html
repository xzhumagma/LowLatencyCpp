37 auto addFill(const Exchange::MEClientResponse *client_response, Logger *logger) noexcept {
38 const auto old_position = position_;
39 const auto side_index = sideToIndex(client_response->side_);
40 const auto opp_side_index = sideToIndex(client_response->side_ == Side::BUY ? Side::SELL : Side::BUY);
41 const auto side_value = sideToValue(client_response->side_);
42 position_ += client_response->exec_qty_ * side_value;
43 volume_ += client_response->exec_qty_;
44
45 if (old_position * sideToValue(client_response->side_) >= 0) { // opened / increased position.
46 open_vwap_[side_index] += (client_response->price_ * client_response->exec_qty_);
47 } else { // decreased position.
48 const auto opp_side_vwap = open_vwap_[opp_side_index] / std::abs(old_position);
49 open_vwap_[opp_side_index] = opp_side_vwap * std::abs(position_);
50 real_pnl_ += std::min(static_cast<int32_t>(client_response->exec_qty_), std::abs(old_position)) *
51 (opp_side_vwap - client_response->price_) * sideToValue(client_response->side_);
52 if (position_ * old_position < 0) { // flipped position to opposite sign.
53 open_vwap_[side_index] = (client_response->price_ * std::abs(position_));
54 open_vwap_[opp_side_index] = 0;
55 }
56 }
57
58 if (!position_) { // flat
59 open_vwap_[sideToIndex(Side::BUY)] = open_vwap_[sideToIndex(Side::SELL)] = 0;
60 unreal_pnl_ = 0;
61 } else {
62 if (position_ > 0)
63 unreal_pnl_ =
64 (client_response->price_ - open_vwap_[sideToIndex(Side::BUY)] / std::abs(position_)) *
65 std::abs(position_);
66 else
67 unreal_pnl_ =
68 (open_vwap_[sideToIndex(Side::SELL)] / std::abs(position_) - client_response->price_) *
69 std::abs(position_);
70 }
71
72 total_pnl_ = unreal_pnl_ + real_pnl_;
73
74 std::string time_str;
75 logger->log("%:% %() % % %\n", __FILE__, __LINE__, __FUNCTION__, Common::getCurrentTimeStr(&time_str),
76 toString(), client_response->toString().c_str());
77 }
