55 auto moveOrder(OMOrder *order, TickerId ticker_id, Price price, Side side, Qty qty) noexcept {
56 switch (order->order_state_) {
57 case OMOrderState::LIVE: {
58 if(order->price_ != price)
59 cancelOrder(order);

61 break;
62 case OMOrderState::INVALID:
63 case OMOrderState::DEAD: {

66 if(LIKELY(risk_result == RiskCheckResult::ALLOWED))
67 newOrder(order, ticker_id, price, side, qty);

69 logger_->log("%:% %() % Ticker:% Side:% Qty:% RiskCheckResult:%\n", __FILE__, __LINE__, __FUNCTION__,
70 Common::getCurrentTimeStr(&time_str_),
71 tickerIdToString(ticker_id), sideToString(side), qtyToString(qty),
72 riskCheckResultToString(risk_result));
73 }
74 }
