48 auto flushQueue() noexcept {
49 while (running_) {
50 // pointer to the logger element
51 // if queue is not empty and next is not a null pointer
52 for (auto next = queue_.getNextToRead(); queue_.size() && next; next = queue_.getNextToRead()) {
53 switch (next->type_) {
54 case LogType::CHAR:
55 file_ << next->u_.c;
56 break;
57 case LogType::INTEGER:

59 break;
60 case LogType::LONG_INTEGER:

63 case LogType::LONG_LONG_INTEGER:
64 file_ << next->u_.ll;
65 break;
66 case LogType::UNSIGNED_INTEGER:

69 case LogType::UNSIGNED_LONG_INTEGER:

72 case LogType::UNSIGNED_LONG_LONG_INTEGER:
73 file_ << next->u_.ull;
74 break;
75 case LogType::FLOAT:

78 case LogType::DOUBLE:

85 // with an ofstream object, it forces the write buffer to "flush" its contents to the underlying file.

107 std::string time_str;
108 std::cerr << Common::getCurrentTimeStr(&time_str) << " Flushing and closing Logger for " << file_name_ << std::endl;
109
110 while (queue_.size()) {
111 std::this_thread::sleep_for(std::chrono::seconds(1));
112 }
113 // pauses this thread for 1 second at a time to avoid busu waiting.
114 running_ = false;
115 // logger_thread_ is joined to ensure it finishes execution before hte logger object is destroyed. This is crucial to avoid
116 // any concurrent access to member variables that are being destroyed or any other undefined behavior.
117 // In c++, when you create a new thread, the thread starts running concurrently with the rest of the program. "joining" a thread means
118 // waiting for that thread to finish its execution.
119 logger_thread_->join();
120
121 file_.close();
122 std::cerr << Common::getCurrentTimeStr(&time_str) << " Logger for " << file_name_ << " exiting." << std::endl;
123 }

177 template<typename T, typename... A>
178 auto log(const char *s, const T &value, A... args) noexcept {
179 while (*s) {
180 if (*s == '%') {
181 if (UNLIKELY(*(s + 1) == '%')) { // to allow %% -> % escape character.
182 ++s;
183 } else {
184 pushValue(value); // substitute % with the value specified in the arguments.
185 log(s + 1, args...); // pop an argument and call self recursively.
186 return;
187 }
188 }
189 pushValue(*s++);
190 }
191 FATAL("extra arguments provided to log()");
192 }
